## Schema Supabase

```sql
CREATE TABLE company_members (
  id TEXT PRIMARY KEY, -- agent ID (ex: 'spfp-marketing') ou user UUID
  user_id UUID REFERENCES auth.users(id), -- owner do registro
  name TEXT NOT NULL,
  avatar TEXT, -- emoji ou URL de imagem
  role TEXT, -- 'agent' | 'human'
  squad TEXT, -- squad de origem
  agent_id TEXT, -- ID do agente AIOS (se role = 'agent')
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT now()
);
```

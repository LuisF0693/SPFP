## Schema Supabase

```sql
CREATE TABLE company_tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  board_id UUID REFERENCES company_boards(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id),
  title TEXT NOT NULL,
  description TEXT,
  status TEXT DEFAULT 'backlog', -- backlog | in_progress | review | done
  priority TEXT DEFAULT 'medium', -- urgent | high | medium | low
  assignee_id TEXT, -- agent ID ou user ID
  assignee_name TEXT,
  assignee_avatar TEXT,
  tags TEXT[],
  due_date DATE,
  sort_order INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);
-- RLS: user_id = auth.uid()
```
